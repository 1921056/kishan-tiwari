









## //program to get average of five numbers\\ **

#include<stdio.h>
 int main()                                           
 {
        int a,b,c,d,e;                                       
        scanf("%d",&a);
        scanf("%d",&b);
        scanf("%d",&c);                                      
        scanf("%d",&d);
        scanf("%d",&e);
        int f=(a+b+c+d+e)/5;                                 
        printf("average of five numbers:%d\n \n",f);
 }
